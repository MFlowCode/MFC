197  SUBROUTINE s_convert_species_to_mixture_variables(q_cons_vf, j,k,l) ! --
198 
199  TYPE(scalar_field), &
200  DIMENSION(sys_size), &
201  INTENT(IN) :: q_cons_vf
202 
203  INTEGER, INTENT(IN) :: j,k,l
204 
205  INTEGER :: i
206 
207 
208  ! Computing the density, the specific heat ratio function and the
209  ! liquid stiffness function, respectively
210  IF(adv_alphan) THEN
211  IF (bubbles .NEQV. .true.) THEN
212  rho_sf(j,k,l) = 0d0
213  gamma_sf(j,k,l) = 0d0
214  pi_inf_sf(j,k,l) = 0d0
215 
216  DO i = 1, num_fluids
217  rho_sf(j,k,l) = rho_sf(j,k,l) &
218  + q_cons_vf( i )%sf(j,k,l)
219  gamma_sf(j,k,l) = gamma_sf(j,k,l) &
220  + q_cons_vf(i+e_idx)%sf(j,k,l) &
221  * fluid_pp(i)%gamma
222  pi_inf_sf(j,k,l) = pi_inf_sf(j,k,l) &
223  + q_cons_vf(i+e_idx)%sf(j,k,l) &
224  * fluid_pp(i)%pi_inf
225  END DO
226  ELSE
227  rho_sf(j,k,l) = q_cons_vf(1)%sf(j,k,l)
228  gamma_sf(j,k,l) = fluid_pp(1)%gamma
229  pi_inf_sf(j,k,l) = fluid_pp(1)%pi_inf
230  END IF
231  ELSE
232  rho_sf(j,k,l) = q_cons_vf(num_fluids)%sf(j,k,l)
233  gamma_sf(j,k,l) = fluid_pp(num_fluids)%gamma
234  pi_inf_sf(j,k,l) = fluid_pp(num_fluids)%pi_inf
235 
236  DO i = 1, num_fluids-1
237  rho_sf(j,k,l) = rho_sf(j,k,l) &
238  + q_cons_vf( i )%sf(j,k,l)
239  gamma_sf(j,k,l) = gamma_sf(j,k,l) &
240  + q_cons_vf(i+e_idx)%sf(j,k,l) &
241  * ( fluid_pp( i )%gamma &
242  - fluid_pp(num_fluids)%gamma )
243  pi_inf_sf(j,k,l) = pi_inf_sf(j,k,l) &
244  + q_cons_vf(i+e_idx)%sf(j,k,l) &
245  * ( fluid_pp( i )%pi_inf &
246  - fluid_pp(num_fluids)%pi_inf )
247  END DO
248 
249  END IF

260 
261  ! Allocating the density, the specific heat ratio function and the
262  ! liquid stiffness function, respectively
263 
264  ! Simulation is at least 2D
265  IF(n > 0) THEN
266 
267  ! Simulation is 3D
268  IF(p > 0) THEN
269 
270  ALLOCATE( rho_sf( -buff_size:m+buff_size, &
271  -buff_size:n+buff_size, &
272  -buff_size:p+buff_size ))
273  ALLOCATE( gamma_sf( -buff_size:m+buff_size, &
274  -buff_size:n+buff_size, &
275  -buff_size:p+buff_size ))
276  ALLOCATE(pi_inf_sf( -buff_size:m+buff_size, &
277  -buff_size:n+buff_size, &
278  -buff_size:p+buff_size ))
279 
280  ! Simulation is 2D
281  ELSE
282 
283  ALLOCATE( rho_sf( -buff_size:m+buff_size, &
284  -buff_size:n+buff_size, &
285  0:0 ))
286  ALLOCATE( gamma_sf( -buff_size:m+buff_size, &
287  -buff_size:n+buff_size, &
288  0:0 ))
289  ALLOCATE(pi_inf_sf( -buff_size:m+buff_size, &
290  -buff_size:n+buff_size, &
291  0:0 ))
292 
293  END IF
294 
295  ! Simulation is 1D
296  ELSE
297 
298  ALLOCATE( rho_sf( -buff_size:m+buff_size, &
299  0:0 , &
300  0:0 ))
301  ALLOCATE( gamma_sf( -buff_size:m+buff_size, &
302  0:0 , &
303  0:0 ))
304  ALLOCATE(pi_inf_sf( -buff_size:m+buff_size, &
305  0:0 , &
306  0:0 ))
307 
308  END IF
309 
310 
311  ! Depending on the model selection for the equations of motion, the
312  ! appropriate procedure for the conversion to the mixture variables
313  ! is targeted by the procedure pointer
314 
315  IF(model_eqns == 1) THEN ! Gamma/pi_inf model
316  s_convert_to_mixture_variables => &
317  s_convert_mixture_to_mixture_variables
318  ELSEIF (bubbles) THEN ! Volume fraction model
319  s_convert_to_mixture_variables => &
320  s_convert_species_to_mixture_variables_bubbles
321  ELSE ! Volume fraction model
322  s_convert_to_mixture_variables => &
323  s_convert_species_to_mixture_variables
324  END IF
325 
326 
327  ! Annotating the dimensionality of the dataset undergoing the post-
328  ! process. A flag value of 1 indicates that the dataset is 3D, while
329  ! a flag value of 0 indicates that it is not.
330  IF(p > 0) THEN
331  flg = 1
332  ELSE
333  flg = 0

345  SUBROUTINE s_convert_conservative_to_primitive_variables( q_cons_vf, &
346  q_prim_vf )
347 
348  TYPE(scalar_field), &
349  DIMENSION(sys_size), &
350  INTENT(IN) :: q_cons_vf
351 
352  TYPE(scalar_field), &
353  DIMENSION(sys_size), &
354  INTENT(INOUT) :: q_prim_vf
355 
356  ! Dynamic pressure, as defined in the incompressible flow sense
357  REAL(KIND(0d0)) :: dyn_pres
358 
359  ! Bubble parameters
360  REAL(KIND(0d0)) :: nbub
361  REAL(KIND(0d0)), dimension(:), allocatable :: nRtmp
362 
363  INTEGER :: i,j,k,l
364 
365  allocate( nrtmp(nb) )
366 
367  ! Converting the conservative variables to the primitive variables
368  DO l = -buff_size*flg, (p+buff_size)*flg
369  DO k = -buff_size, n+buff_size
370  DO j = -buff_size, m+buff_size
371 
372  ! Obtaining the density, specific heat ratio function
373  ! and the liquid stiffness function, respectively
374  CALL s_convert_to_mixture_variables(q_cons_vf, j,k,l)
375 
376  ! Transferring the continuity equation(s) variable(s)
377  DO i = 1, cont_idx%end
378  q_prim_vf(i)%sf(j,k,l) = q_cons_vf(i)%sf(j,k,l)
379  END DO
380 
381  ! Zeroing out the dynamic pressure since it is computed
382  ! iteratively by cycling through the momentum equations
383  dyn_pres = 0d0
384 
385  ! Computing velocity and dynamic pressure from momenta
386  DO i = mom_idx%beg, mom_idx%end
387  IF (model_eqns==4) THEN
388  !u = \rho u/ \rho
389  q_prim_vf(i)%sf(j,k,l) = q_cons_vf(i)%sf(j,k,l) / &
390  q_cons_vf(1)%sf(j,k,l)
391  ELSE
392  q_prim_vf(i)%sf(j,k,l) = q_cons_vf(i)%sf(j,k,l) / &
393  rho_sf(j,k,l)
394  dyn_pres = dyn_pres + q_cons_vf(i)%sf(j,k,l) * &
395  q_prim_vf(i)%sf(j,k,l) / 2d0
396  END IF
397  END DO
398 
399  IF (model_eqns == 4) THEN
400  ! Computing the pressure from the energy
401  ! Tait EOS
402  ! p_l = (pl0 + pi_infty)(rho/(rho_l0(1-alf)))^gamma
403  ! - pi_infty
404  q_prim_vf(e_idx)%sf(j,k,l) = &
405  (pref+fluid_pp(1)%pi_inf) * &
406  (( &
407  q_cons_vf(1)%sf(j,k,l)/ &
408  (rhoref*(1.d0-q_cons_vf(alf_idx)%sf(j,k,l))) &
409  ) ** (1.d0/fluid_pp(1)%gamma + 1.d0)) - fluid_pp(1)%pi_inf
410  ELSE IF ( (model_eqns .NE. 4) .AND. bubbles .NEQV. .true. ) THEN
411  ! Computing the pressure from the energy
412  q_prim_vf(e_idx)%sf(j,k,l) = &
413  ( q_cons_vf(e_idx)%sf(j,k,l) &
414  - dyn_pres -pi_inf_sf(j,k,l) ) / gamma_sf(j,k,l)
415  ELSE
416  ! p = ( E/(1-alf) - 0.5 rho u u/(1-alf) - pi_inf_k )/gamma_k
417  q_prim_vf(e_idx)%sf(j,k,l) = &
418  ( (q_cons_vf(e_idx)%sf(j,k,l) &
419  - dyn_pres) / (1.d0 - q_cons_vf(alf_idx)%sf(j,k,l)) &
420  - pi_inf_sf(j,k,l) &
421  ) / gamma_sf(j,k,l)
422  END IF
423 
424  ! Set partial pressures to mixture pressure
425  IF(model_eqns == 3) THEN
426  DO i = internalenergies_idx%beg, internalenergies_idx%end
427  q_prim_vf(i)%sf(j,k,l) = q_prim_vf(e_idx)%sf(j,k,l)
428  END DO
429  END IF
430 
431  ! Transferring the advection equation(s) variable(s)
432  DO i = adv_idx%beg, adv_idx%end
433  q_prim_vf(i)%sf(j,k,l) = q_cons_vf(i)%sf(j,k,l)
434  END DO
435 
436  ! \phi = (n\phi)/n (n = nbub)
437  IF (bubbles) THEN
438  ! n = sqrt( 4pi/(3 alpha) * (nR)**3 )
439  DO i = 1,nb
440  nrtmp(i) = q_cons_vf(bub_idx%rs(i))%sf(j,k,l)
441  END DO
442  CALL s_comp_n_from_cons( q_cons_vf(alf_idx)%sf(j,k,l), nrtmp, nbub)
443  DO i = bub_idx%beg, sys_size
444  q_prim_vf(i)%sf(j,k,l) = q_cons_vf(i)%sf(j,k,l)/nbub
445  END DO
446  END IF
447  END DO
448  END DO
449  END DO
450 
451  DEALLOCATE( nrtmp )
452 
453  END SUBROUTINE s_convert_conservative_to_primitive_variables ! ---------
