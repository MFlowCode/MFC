516  SUBROUTINE s_open_formatted_database_file(t_step) ! --------------------
517 
518  INTEGER, INTENT(IN) :: t_step
519 
520  CHARACTER(LEN = LEN_TRIM(case_dir) + 3*name_len) :: file_loc
522 
523 
524  ! Silo-HDF5 Database Format ========================================
525 
526  IF(format == 1) THEN
527 
528  ! Generating the relative path to the formatted database slave
529  ! file, that is to be opened for the current time-step, t_step
530  WRITE(file_loc, '(A,I0,A)') '/', t_step, '.silo'
531  file_loc = trim(proc_rank_dir) // trim(file_loc)
532 
533  ! Creating formatted database slave file at the above location
534  ! and setting up the structure of the file and its header info
535  ierr = dbcreate( trim(file_loc), len_trim(file_loc), &
536  db_clobber, db_local, 'MFC v3.0', 8, &
537  db_hdf5, dbfile )
538 
539  ! Verifying that the creation and setup process of the formatted
540  ! database slave file has been performed without errors. If this
541  ! is not the case, the post-process exits.
542  IF(dbfile == -1) THEN
543  print '(A)', 'Unable to create Silo-HDF5 database ' // &
544  'slave file '// trim(file_loc) // '. ' // &
545  'Exiting ...'
546  CALL s_mpi_abort()
547  END IF
548 
549  ! Next, analogous steps to the ones above are carried out by the
550  ! root process to create and setup the formatted database master
551  ! file.
552  IF(proc_rank == 0) THEN
553 
554  WRITE(file_loc, '(A,I0,A)') '/collection_', t_step, '.silo'
555  file_loc = trim(rootdir) // trim(file_loc)
556 
557  ierr = dbcreate( trim(file_loc), len_trim(file_loc), &
558  db_clobber, db_local, 'MFC v3.0', 8, &
559  db_hdf5, dbroot )
560 
561  IF(dbroot == -1) THEN
562  print '(A)', 'Unable to create Silo-HDF5 database ' // &
563  'master file '// trim(file_loc) //'. ' // &
564  'Exiting ...'
565  CALL s_mpi_abort()
566  END IF
567 
568  END IF
569 
570  ! ==================================================================
571 
572 
573  ! Binary Database Format ===========================================
574 
575  ELSE
576 
577  ! Generating the relative path to the formatted database slave
578  ! file, that is to be opened for the current time-step, t_step
579  WRITE(file_loc, '(A,I0,A)') '/', t_step, '.dat'
580  file_loc = trim(proc_rank_dir) // trim(file_loc)
581 
582  ! Creating the formatted database slave file, at the previously
583  ! precised relative path location, and setting up its structure
584  OPEN( dbfile, iostat = err, file = trim(file_loc), &
585  form = 'unformatted', status = 'replace' )
586 
587  ! Verifying that the creation and setup process of the formatted
588  ! database slave file has been performed without errors. If this
589  ! is not the case, the post-process exits.
590  IF(err /= 0) THEN
591  print '(A)', 'Unable to create Binary database slave ' // &
592  'file '// trim(file_loc)//'. Exiting ...'
593  CALL s_mpi_abort()
594  END IF
595 
596  ! Further defining the structure of the formatted database slave
597  ! file by describing in it the dimensionality of post-processed
598  ! data as well as the total number of flow variable(s) that will
599  ! eventually be stored in it
600  WRITE(dbfile) m, n, p, dbvars
601 
602  ! Next, analogous steps to the ones above are carried out by the
603  ! root process to create and setup the formatted database master
604  ! file. Note that this is only done in multidimensional cases.
605  IF(n == 0 .AND. proc_rank == 0) THEN
606 
607  WRITE(file_loc, '(A,I0,A)') '/', t_step, '.dat'
608  file_loc = trim(rootdir) // trim(file_loc)
609 
610  OPEN( dbroot, iostat = err, file = trim(file_loc), &
611  form = 'unformatted', status = 'replace' )
612 
613  IF(err /= 0) THEN
614  print '(A)', 'Unable to create Binary database ' // &
615  'master file ' // trim(file_loc) // &
616  '. Exiting ...'
617  CALL s_mpi_abort()
618  END IF
619 
620  WRITE(dbroot) m_root, 0, 0, dbvars
621 
622  END IF
623 
624  END IF
625 
626  ! END: Binary Database Format ======================================
627 
628 
629  END SUBROUTINE s_open_formatted_database_file ! ------------------------

653  SUBROUTINE s_write_grid_to_formatted_database_file(t_step) ! -----------
654 
655  INTEGER, INTENT(IN) :: t_step
656 
662  CHARACTER(LEN = 4*name_len), DIMENSION(num_procs) :: meshnames
663  INTEGER , DIMENSION(num_procs) :: meshtypes
665 
666  INTEGER :: i
667 
668 
669  ! Silo-HDF5 Database Format ========================================
670 
671  IF(format == 1 .AND. n > 0) THEN
672 
673  ! For multidimensional data sets, the spatial extents of all of
674  ! the grid(s) handled by the local processor(s) are recorded so
675  ! that they may be written, by root processor, to the formatted
676  ! database master file.
677  IF(num_procs > 1) THEN
678  CALL s_mpi_gather_spatial_extents(spatial_extents)
679 
680  ELSEIF(p > 0) THEN
681  IF (grid_geometry == 3) THEN
682  spatial_extents(:,0) = (/ minval(y_cb), minval(z_cb), &
683  minval(x_cb), maxval(y_cb), &
684  maxval(z_cb), maxval(x_cb) /)
685  ELSE
686  spatial_extents(:,0) = (/ minval(x_cb), minval(y_cb), &
687  minval(z_cb), maxval(x_cb), &
688  maxval(y_cb), maxval(z_cb) /)
689  END IF
690 
691  ELSE
692  spatial_extents(:,0) = (/ minval(x_cb), minval(y_cb), &
693  maxval(x_cb), maxval(y_cb) /)
694 
695  END IF
696 
697  ! Next, the root processor proceeds to record all of the spatial
698  ! extents in the formatted database master file. In addition, it
699  ! also records a sub-domain connectivity map so that the entire
700  ! grid may be reassembled by looking at the master file.
701  IF(proc_rank == 0) THEN
702 
703  DO i = 1, num_procs
704  WRITE(meshnames(i), '(A,I0,A,I0,A)') '../p', i-1, &
705  '/', t_step, '.silo:rectilinear_grid'
706  END DO
707 
708  meshtypes = db_quad_rect
709 
710  err = dbset2dstrlen(len(meshnames(1)))
711  err = dbmkoptlist(2, optlist)
712  err = dbaddiopt( optlist, dbopt_extents_size, &
713  SIZE(spatial_extents,1) )
714  err = dbadddopt(optlist, dbopt_extents, spatial_extents)
715  err = dbputmmesh( dbroot, 'rectilinear_grid', 16, &
716  num_procs, meshnames, &
717  len_trim(meshnames), &
718  meshtypes, optlist, ierr )
719  err = dbfreeoptlist(optlist)
720 
721  END IF
722 
723  ! Finally, the local quadrilateral mesh, either 2D or 3D, along
724  ! with its offsets that indicate the presence and size of ghost
725  ! zone layer(s), are put in the formatted database slave file.
726 
727  IF (coarsen_silo) THEN
728  coarse_x_cb(-1-offset_x%beg : -1) = x_cb(-1-offset_x%beg : -1)
