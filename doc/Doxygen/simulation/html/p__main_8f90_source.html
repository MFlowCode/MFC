65 
66  USE m_cbc !< characteristic boundary conditions (cbc)

81  INTEGER :: t_step
82 
83  CALL system_clock(count=cpu_start, count_rate=cpu_rate)
84 
85  ! Initializing MPI execution environment
86  CALL s_mpi_initialize()
87 
88 
89 
90  ! The rank 0 processor assigns default values to the user inputs prior to
91  ! reading them in from the input file. Next, the user inputs are read and
92  ! their consistency is checked. The identification of any inconsistencies
93  ! will result in the termination of the simulation.
94  IF(proc_rank == 0) THEN
95  CALL s_assign_default_values_to_user_inputs()
96  CALL s_read_input_file()
97  CALL s_check_input_file()
98  END IF
99 
100  ! Broadcasting the user inputs to all of the processors and performing the
101  ! parallel computational domain decomposition. Neither procedure has to be
102  ! carried out if the simulation is in fact not truly executed in parallel.
103  CALL s_mpi_bcast_user_inputs()
104  CALL s_initialize_parallel_io()
105  CALL s_mpi_decompose_computational_domain()
106 
107  ! Computation of parameters, allocation of memory, association of pointers,
108  ! and/or the execution of any other tasks needed to properly configure the
109  ! modules. The preparations below DO NOT DEPEND on the grid being complete.
110  CALL s_initialize_global_parameters_module()
111  CALL s_initialize_mpi_proxy_module()
112  IF (grid_geometry == 3) CALL s_initialize_fftw_module()
113  CALL s_initialize_variables_conversion_module()
114  CALL s_initialize_start_up_module()
115  CALL s_initialize_riemann_solvers_module()
116  CALL s_initialize_rhs_module()
117  CALL s_initialize_data_output_module()
118  CALL s_initialize_derived_variables_module()
119  CALL s_initialize_time_steppers_module()
120 
121  ! Associate pointers for serial or parallel I/O
122  IF (parallel_io .NEQV. .true.) THEN
123  s_read_data_files => s_read_serial_data_files
124  s_write_data_files => s_write_serial_data_files
125  ELSE
126  s_read_data_files => s_read_parallel_data_files
127  s_write_data_files => s_write_parallel_data_files
128  END IF
129 
130  ! Reading in the user provided initial condition and grid data
131  CALL s_read_data_files(q_cons_ts(1)%vf)
132  IF (model_eqns == 3) CALL s_initialize_internal_energy_equations(q_cons_ts(1)%vf)
133 
134  ! Populating the buffers of the grid variables using the boundary conditions
135  CALL s_populate_grid_variables_buffers()
136 
137  CALL s_populate_variables_buffers(q_cons_ts(1)%vf)
138  IF (we_size > 0 .AND. (we_riemann_flux .OR. we_rhs_flux)) CALL s_account_for_capillary_potential_energy(q_cons_ts(1)%vf)
139 
140  ! Computation of parameters, allocation of memory, association of pointers,
141  ! and/or execution of any other tasks that are needed to properly configure
142  ! the modules. The preparations below DO DEPEND on the grid being complete.
143  CALL s_initialize_weno_module()
144  CALL s_initialize_cbc_module()
145 
146  CALL s_initialize_derived_variables()
147 
148  ! Setting the time-step iterator to the first time-step
149  t_step = t_step_start
150 
151  ! Time-stepping Loop =======================================================
152  DO
153  IF (proc_rank==0 .AND. mod(t_step,t_step_save)==0 ) THEN
154  print*, 'Time step ', t_step, ' of ', t_step_stop
155  END IF
156 
157  ! Total-variation-diminishing (TVD) Runge-Kutta (RK) time-steppers
158  IF(time_stepper == 1) THEN
159  CALL s_1st_order_tvd_rk(t_step)
160  ELSEIF(time_stepper == 2) THEN
161  CALL s_2nd_order_tvd_rk(t_step)
162  ELSEIF(time_stepper == 3) THEN
163  CALL s_3rd_order_tvd_rk(t_step)
164  ELSEIF(time_stepper == 4) THEN
165  CALL s_4th_order_rk(t_step)
166  ELSE
167  CALL s_5th_order_rk(t_step)
168  END IF
169 
170  CALL s_compute_derived_variables(t_step)
171 
172  ! Time-stepping loop controls
173  IF(t_step == t_step_stop) THEN
174  EXIT
175  ELSE
176  t_step = t_step + 1
177  END IF
178 
179  ! Backing up the grid and conservative variables data
180  IF(mod(t_step-t_step_start, t_step_save) == 0) THEN
181  CALL s_write_data_files(q_cons_ts(1)%vf, t_step)
182  END IF
183 
184  CALL system_clock(cpu_end)
185  END DO
186  ! ==========================================================================
187 
188  ! Disassociate pointers for serial and parallel I/O
189  s_read_data_files => null()
190  s_write_data_files => null()
191 
192  ! Deallocation and/or disassociation procedures for the modules
193  CALL s_finalize_time_steppers_module()
194  CALL s_finalize_derived_variables_module()
195  CALL s_finalize_data_output_module()
196  CALL s_finalize_rhs_module()
197  CALL s_finalize_cbc_module()
198  CALL s_finalize_riemann_solvers_module()
199  CALL s_finalize_weno_module()
200  CALL s_finalize_start_up_module()
201  CALL s_finalize_variables_conversion_module()
202  IF (grid_geometry == 3) CALL s_finalize_fftw_module()
203  CALL s_finalize_mpi_proxy_module()
204  CALL s_finalize_global_parameters_module()
205 
206 
207  ! Terminating MPI execution environment
208  CALL s_mpi_finalize()
209 
210 
211 END PROGRAM p_main
