64  ! Initialization of the MPI environment
65  CALL s_mpi_initialize()
66 
67 
68  ! Rank 0 processor assigns default values to user inputs prior to reading
69  ! those in from the input file. Next, the user inputs are read in and their
70  ! consistency is checked. The detection of any inconsistencies automatically
71  ! leads to the termination of the pre-process.
72  IF (proc_rank == 0) THEN
73  CALL s_assign_default_values_to_user_inputs()
74  CALL s_read_input_file()
75  CALL s_check_input_file()
76  END IF
77 
78 
79  ! Broadcasting the user inputs to all of the processors and performing the
80  ! parallel computational domain decomposition. Neither procedure has to be
81  ! carried out if pre-process is in fact not truly executed in parallel.
82  CALL s_mpi_bcast_user_inputs()
83  CALL s_initialize_parallel_io()
84  CALL s_mpi_decompose_computational_domain()
85 
86  ! Computation of parameters, allocation procedures, and/or any other tasks
87  ! needed to properly setup the modules
88  CALL s_initialize_global_parameters_module()
89  CALL s_initialize_data_output_module()
90  CALL s_initialize_variables_conversion_module()
91  CALL s_initialize_start_up_module()
92  CALL s_initialize_grid_module()
93  CALL s_initialize_initial_condition_module()
94 
95  ! Associate pointers for serial or parallel I/O
96  IF (parallel_io .NEQV. .true.) THEN
97  s_generate_grid => s_generate_serial_grid
98  s_read_grid_data_files => s_read_serial_grid_data_files
99  s_read_ic_data_files => s_read_serial_ic_data_files
100  s_write_data_files => s_write_serial_data_files
101  ELSE
102  s_generate_grid => s_generate_parallel_grid
103  s_read_grid_data_files => s_read_parallel_grid_data_files
104  s_read_ic_data_files => s_read_parallel_ic_data_files
105  s_write_data_files => s_write_parallel_data_files
106  END IF
107 
108 
109  ! Setting up the grid and the initial condition. If the grid is read in from
110  ! preexisting grid data files, it is checked for consistency. If the grid is
111  ! not read in, it is generated from scratch according to the inputs provided
112  ! by the user. The initial condition may also be read in. It in turn is not
113  ! checked for consistency since it WILL further be edited by the pre-process
114  ! and also because it may be incomplete at the time it is read in. Finally,
115  ! when the grid and initial condition are completely setup, they are written
116  ! to their respective data files.
117 
118  ! Setting up grid and initial condition
119 
120  IF(old_grid) THEN
121  CALL s_read_grid_data_files(dflt_int)
122  CALL s_check_grid_data_files()
123  ELSE
124  IF (parallel_io .NEQV. .true.) THEN
125  CALL s_generate_grid(dflt_int)
126  ELSE
127  IF (proc_rank == 0) CALL s_generate_grid(dflt_int)
128  CALL s_mpi_barrier()
129  CALL s_read_grid_data_files(dflt_int)
130  CALL s_check_grid_data_files()
131  END IF
132  END IF
133 
134  IF(old_ic) CALL s_read_ic_data_files(q_cons_vf)
135 
136  CALL s_generate_initial_condition()
137 
138  CALL s_write_data_files(q_cons_vf)
139 
140  ! Disassociate pointers for serial and parallel I/O
141  s_generate_grid => null()
142  s_read_grid_data_files => null()
143  s_read_ic_data_files => null()
144  s_write_data_files => null()
145 
146  ! Deallocation procedures for the modules
147  CALL s_finalize_initial_condition_module()
148  CALL s_finalize_grid_module()
149  CALL s_finalize_start_up_module()
150  CALL s_finalize_variables_conversion_module()
151  CALL s_finalize_data_output_module()
152  CALL s_finalize_global_parameters_module()
153 
154  ! Finalization of the MPI environment
155  CALL s_mpi_finalize()
156 
157 END PROGRAM p_main
